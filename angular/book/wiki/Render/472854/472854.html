Category: Recursion
Description: Declaring a simple recursive function
Code: 
let SampleRec1() =
    let rec fib n = if n < 2 then 1 else fib (n-1) + fib (n-2) 
    for i = 1 to 10 do 
        printfn "fib %d = %d" i (fib i)
