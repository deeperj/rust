let defineEvent() = 
  let e = new Event()
  //trigger event, but not subscriber. Nothing happens.
  e.Trigger(5)
  //subscribe to the event
  Event.add (fun n->printfn "%d" n) e.Publish
  //trigger the event again
  e.Trigger(10);
